üî∑ 3. Smoothing (—Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ)

–ü—Ä–µ–¥–æ—Ç–≤—Ä–∞—â–µ–Ω–∏–µ overfitting:

def target_encode_with_smoothing(df, category_col, target_col, alpha=10):
    # –ì–ª–æ–±–∞–ª—å–Ω–æ–µ —Å—Ä–µ–¥–Ω–µ–µ
    global_mean = df[target_col].mean()
    
    # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –∫–∞—Ç–µ–≥–æ—Ä–∏—è–º
    agg = df.groupby(category_col)[target_col].agg(['sum', 'count'])
    
    # Smoothed mean
    # weight = n / (n + alpha)
    smoothed_mean = (agg['sum'] + global_mean * alpha) / (agg['count'] + alpha)
    
    return smoothed_mean

# –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ
smoothed_encoding = target_encode_with_smoothing(df, 'category', 'target', alpha=10)
df['category_smoothed'] = df['category'].map(smoothed_encoding)

# alpha - –≥–∏–ø–µ—Ä–ø–∞—Ä–∞–º–µ—Ç—Ä:
# alpha –±–æ–ª—å—à–æ–µ ‚Üí –±–ª–∏–∂–µ –∫ global mean (–º–µ–Ω—å—à–µ overfitting)

üî∑ 4. Cross-validation Target Encoding

–ü—Ä–∞–≤–∏–ª—å–Ω—ã–π —Å–ø–æ—Å–æ–± –∏–∑–±–µ–∂–∞—Ç—å —É—Ç–µ—á–∫–∏:

from sklearn.model_selection import KFold

def cv_target_encode(X, y, col, n_splits=5, alpha=10):
    kf = KFold(n_splits=n_splits, shuffle=True, random_state=42)
    encoded = np.zeros(len(X))
    global_mean = y.mean()
    
    for train_idx, val_idx in kf.split(X):
        # –í—ã—á–∏—Å–ª—è–µ–º encoding —Ç–æ–ª—å–∫–æ –Ω–∞ train
        X_train, y_train = X.iloc[train_idx], y.iloc[train_idx]
        
        # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –Ω–∞ train
        agg = pd.DataFrame({
            'sum': X_train.groupby(col)[col].count() * 
                   y_train.groupby(X_train[col]).mean(),
            'count': X_train.groupby(col)[col].count()
        })
        
        # Smoothed encoding
        smoothed = (agg['sum'] + global_mean * alpha) / (agg['count'] + alpha)
        
        # –ü—Ä–∏–º–µ–Ω—è–µ–º –∫ validation
        encoded[val_idx] = X.iloc[val_idx][col].map(smoothed).fillna(global_mean)
    
    return encoded

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
X_train['category_encoded'] = cv_target_encode(
    X_train, y_train, 'category', n_splits=5
)

# –†—É—á–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è LOO
def loo_encode(df, cat_col, target_col):
    encoded = np.zeros(len(df))
    
    for idx in range(len(df)):
        # –í—Å–µ –∫—Ä–æ–º–µ —Ç–µ–∫—É—â–µ–≥–æ
        mask = df.index != df.index[idx]
        category = df.loc[df.index[idx], cat_col]
        
        # Mean –±–µ–∑ —Ç–µ–∫—É—â–µ–≥–æ –ø—Ä–∏–º–µ—Ä–∞
        same_category = (df.loc[mask, cat_col] == category)
        if same_category.sum() > 0:
            encoded[idx] = df.loc[mask & same_category, target_col].mean()
        else:
            encoded[idx] = df[target_col].mean()
    
    return encoded
