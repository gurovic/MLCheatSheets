def visualize_optics(X, optics, labels):
    """–ü–æ–ª–Ω–∞—è –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è OPTICS"""
    
    fig = plt.figure(figsize=(16, 5))
    
    # 1. –î–∞–Ω–Ω—ã–µ —Å –∫–ª–∞—Å—Ç–µ—Ä–∞–º–∏
    ax1 = fig.add_subplot(131)
    ax1.scatter(X[:, 0], X[:, 1], c=labels, 
                cmap='viridis', s=10)
    ax1.set_title('–ö–ª–∞—Å—Ç–µ—Ä—ã')
    
    # 2. Reachability plot
    ax2 = fig.add_subplot(132)
    reachability = optics.reachability_[optics.ordering_]
    colors = labels[optics.ordering_]
    
    for i, (r, c) in enumerate(zip(reachability, colors)):
        color = 'black' if c == -1 else plt.cm.viridis(c / max(colors))
        ax2.bar(i, r, color=color, width=1)
    
    ax2.set_xlabel('–ü–æ—Ä—è–¥–æ–∫ —Ç–æ—á–µ–∫')
    ax2.set_ylabel('Reachability')
    ax2.set_title('Reachability Plot')
    
    # 3. Core distances
    ax3 = fig.add_subplot(133)
    core_dist = optics.core_distances_
    valid = core_dist != np.inf
    ax3.hist(core_dist[valid], bins=50, edgecolor='black')
    ax3.set_xlabel('Core Distance')
    ax3.set_ylabel('–ß–∞—Å—Ç–æ—Ç–∞')
    ax3.set_title('Core Distances')
    
    plt.tight_layout()
    plt.show()

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
visualize_optics(X_scaled, optics, labels)
