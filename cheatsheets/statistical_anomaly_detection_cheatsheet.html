üî∑ 2. Z-score –º–µ—Ç–æ–¥

–§–æ—Ä–º—É–ª–∞: z = (x - Œº) / œÉ

import numpy as np
import pandas as pd

def z_score_anomalies(data, threshold=3):
    mean = np.mean(data)
    std = np.std(data)
    z_scores = np.abs((data - mean) / std)
    return z_scores > threshold

# –ü—Ä–∏–º–µ—Ä

    (df['value'] - df['value'].mean()) / df['value'].std()
)

üî∑ 3. Modified Z-score (MAD)

–£—Å—Ç–æ–π—á–∏–≤ –∫ –≤—ã–±—Ä–æ—Å–∞–º, –∏—Å–ø–æ–ª—å–∑—É–µ—Ç –º–µ–¥–∏–∞–Ω—É:

–§–æ—Ä–º—É–ª–∞: M_i = 0.6745 * (x_i - median(x)) / MAD

–≥–¥–µ MAD = median(|x_i - median(x)|)

def modified_z_score_anomalies(data, threshold=3.5):
    median = np.median(data)
    mad = np.median(np.abs(data - median))
    modified_z_scores = 0.6745 * (data - median) / mad
    return np.abs(modified_z_scores) > threshold

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

print(f"–ê–Ω–æ–º–∞–ª–∏–π (MAD): {anomalies.sum()}")

üî∑ 4. IQR –º–µ—Ç–æ–¥

def iqr_anomalies(data, factor=1.5):
    Q1 = np.percentile(data, 25)
    Q3 = np.percentile(data, 75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - factor * IQR
    upper_bound = Q3 + factor * IQR
    
    return (data < lower_bound) | (data > upper_bound)

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

üî∑ 5. Grubbs —Ç–µ—Å—Ç

–ü–∞—Ä–∞–º–µ—Ç—Ä–∏—á–µ—Å–∫–∏–π —Ç–µ—Å—Ç –¥–ª—è –æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏—è –æ–¥–Ω–æ–≥–æ –≤—ã–±—Ä–æ—Å–∞:

from scipy import stats

def grubbs_test(data, alpha=0.05):
    n = len(data)
    mean = np.mean(data)
    std = np.std(data)
    
    # G-—Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
    G = np.max(np.abs(data - mean)) / std
    
    # –ö—Ä–∏—Ç–∏—á–µ—Å–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    t_dist = stats.t.ppf(1 - alpha / (2 * n), n - 2)
    G_critical = ((n - 1) / np.sqrt(n)) *                  np.sqrt(t_dist**2 / (n - 2 + t_dist**2))
    
    return G > G_critical, G, G_critical

is_outlier, G, G_crit = grubbs_test(data)
print(f"Outlier detected: {is_outlier}")
print(f"G: {G:.4f}, Critical: {G_crit:.4f}")

def dixon_test(data, alpha=0.05):
    n = len(data)
    data_sorted = np.sort(data)
    
    # Q-—Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
    if n <= 7:
        Q = (data_sorted[1] - data_sorted[0]) /             (data_sorted[-1] - data_sorted[0])
    elif n <= 10:
        Q = (data_sorted[1] - data_sorted[0]) /             (data_sorted[-2] - data_sorted[0])
    else:
        Q = (data_sorted[2] - data_sorted[0]) /             (data_sorted[-2] - data_sorted[0])
    
    # –ö—Ä–∏—Ç–∏—á–µ—Å–∫–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è (–¥–ª—è alpha=0.05)
    critical_values = {
        7: 0.507, 8: 0.554, 9: 0.512, 10: 0.477
    }
    
    Q_critical = critical_values.get(n, 0.5)
    return Q > Q_critical, Q, Q_critical

# –ü—Ä–∏–º–µ—Ä –Ω–∞ –º–∞–ª–æ–π –≤—ã–±–æ—Ä–∫–µ
small_data = np.array([1.2, 1.5, 1.4, 1.3, 1.6, 5.0])
is_outlier, Q, Q_crit = dixon_test(small_data)
print(f"Outlier: {is_outlier}, Q: {Q:.3f}")

from scipy.spatial.distance import mahalanobis
from scipy.stats import chi2

def mahalanobis_anomalies(data, threshold_p=0.95):
    mean = np.mean(data, axis=0)
    cov = np.cov(data.T)
    inv_cov = np.linalg.inv(cov)
    
    distances = []
    for x in data:
        d = mahalanobis(x, mean, inv_cov)
        distances.append(d)
    
    distances = np.array(distances)
    
    # Chi-squared –ø–æ—Ä–æ–≥
    threshold = chi2.ppf(threshold_p, df=data.shape[1])
    
    return distances**2 > threshold, distances

# 1. –°–∫–æ–ª—å–∑—è—â–µ–µ —Å—Ä–µ–¥–Ω–µ–µ + std
def ts_anomalies(data, window=20, n_std=3):
    rolling_mean = pd.Series(data).rolling(
        window=window
    ).mean()
    rolling_std = pd.Series(data).rolling(
        window=window
    ).std()
    
    upper_bound = rolling_mean + n_std * rolling_std
    lower_bound = rolling_mean - n_std * rolling_std
    
    return (data > upper_bound) | (data < lower_bound)

üî∑ 9. GESD (Generalized ESD)

–û–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö –≤—ã–±—Ä–æ—Å–æ–≤:

def generalized_esd(data, max_outliers=10, 
                   alpha=0.05):
    outliers = []
    data_copy = data.copy()
    
    for i in range(max_outliers):
        n = len(data_copy)
        mean = np.mean(data_copy)
        std = np.std(data_copy, ddof=1)
        
        # Test statistic
        R = np.max(np.abs(data_copy - mean)) / std
        
        # Critical value
        p = 1 - alpha / (2 * (n - i))
        t = stats.t.ppf(p, n - i - 2)
        lambda_critical = (n - i - 1) * t /             np.sqrt((n - i - 2 + t**2) * (n - i))
        
        if R > lambda_critical:
            max_idx = np.argmax(np.abs(data_copy - mean))
            outliers.append(data_copy[max_idx])
            data_copy = np.delete(data_copy, max_idx)
        else:
            break
    
    return outliers

outliers = generalized_esd(data, max_outliers=5)

def histogram_anomalies(data, bins=50, 
                        threshold_percentile=1):
    hist, bin_edges = np.histogram(data, bins=bins)
    
    # –ù–∞–π—Ç–∏ bins —Å –º–∞–ª–æ–π —á–∞—Å—Ç–æ—Ç–æ–π
    threshold = np.percentile(hist, threshold_percentile)
    
    anomalies = np.zeros(len(data), dtype=bool)
    for i, value in enumerate(data):
        bin_idx = np.searchsorted(bin_edges[:-1], value)
        if bin_idx < len(hist) and hist[bin_idx] < threshold:
            anomalies[i] = True
    
    return anomalies

anomalies = histogram_anomalies(data)
print(f"Histogram –∞–Ω–æ–º–∞–ª–∏–π: {anomalies.sum()}")

from scipy.stats import gaussian_kde

def kde_anomalies(data, threshold_percentile=5):
    # KDE
