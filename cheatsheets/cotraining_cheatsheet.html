import numpy as np
from sklearn.naive_bayes import GaussianNB
from sklearn.model_selection import train_test_split

class CoTraining:
    def __init__(self, clf1, clf2, k=5, max_iter=10):
        self.clf1 = clf1  # –ú–æ–¥–µ–ª—å –¥–ª—è view 1
        self.clf2 = clf2  # –ú–æ–¥–µ–ª—å –¥–ª—è view 2
        self.k = k        # –ü—Ä–∏–º–µ—Ä–æ–≤ –¥–ª—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è –Ω–∞ –∏—Ç–µ—Ä–∞—Ü–∏—é
        self.max_iter = max_iter
    
    def fit(self, X1_labeled, X2_labeled, y_labeled,
            X1_unlabeled, X2_unlabeled):
        
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
        X1_l, X2_l, y_l = X1_labeled.copy(), X2_labeled.copy(), y_labeled.copy()
        X1_u, X2_u = X1_unlabeled.copy(), X2_unlabeled.copy()
        
        for iteration in range(self.max_iter):
            # –û–±—É—á–µ–Ω–∏–µ –Ω–∞ —Ç–µ–∫—É—â–∏—Ö —Ä–∞–∑–º–µ—á–µ–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
            self.clf1.fit(X1_l, y_l)
            self.clf2.fit(X2_l, y_l)
            
            # –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –Ω–∞ –Ω–µ—Ä–∞–∑–º–µ—á–µ–Ω–Ω—ã—Ö
            proba1 = self.clf1.predict_proba(X1_u)
            proba2 = self.clf2.predict_proba(X2_u)
            
            # –í—ã–±–æ—Ä –Ω–∞–∏–±–æ–ª–µ–µ —É–≤–µ—Ä–µ–Ω–Ω—ã—Ö
            conf1 = np.max(proba1, axis=1)
            conf2 = np.max(proba2, axis=1)
            
            # –¢–æ–ø-k –æ—Ç –∫–∞–∂–¥–æ–π –º–æ–¥–µ–ª–∏
            idx1 = np.argsort(conf1)[-self.k:]
            idx2 = np.argsort(conf2)[-self.k:]
            
            # –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –æ–±—É—á–∞—é—â—É—é –≤—ã–±–æ—Ä–∫—É
            X1_l = np.vstack([X1_l, X1_u[idx2]])
            X2_l = np.vstack([X2_l, X2_u[idx2]])
            y_l = np.hstack([y_l, self.clf2.predict(X2_u[idx2])])
            
            X1_l = np.vstack([X1_l, X1_u[idx1]])
            X2_l = np.vstack([X2_l, X2_u[idx1]])
            y_l = np.hstack([y_l, self.clf1.predict(X1_u[idx1])])
            
            # –£–¥–∞–ª–µ–Ω–∏–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–Ω—ã—Ö –Ω–µ—Ä–∞–∑–º–µ—á–µ–Ω–Ω—ã—Ö
            mask = np.ones(len(X1_u), dtype=bool)
            mask[np.concatenate([idx1, idx2])] = False
            X1_u = X1_u[mask]
            X2_u = X2_u[mask]
            
            if len(X1_u) < self.k:
                break
        
        # –§–∏–Ω–∞–ª—å–Ω–æ–µ –æ–±—É—á–µ–Ω–∏–µ
        self.clf1.fit(X1_l, y_l)
        self.clf2.fit(X2_l, y_l)
        
    def predict(self, X1, X2):
        # –ö–æ–º–±–∏–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π
        pred1 = self.clf1.predict_proba(X1)
        pred2 = self.clf2.predict_proba(X2)
        return np.argmax((pred1 + pred2) / 2, axis=1)

# –ü—Ä–∏–º–µ—Ä —Å margin sampling
def select_by_margin(proba, k):
    """–í—ã–±—Ä–∞—Ç—å k –ø—Ä–∏–º–µ—Ä–æ–≤ —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º margin"""
    sorted_proba = np.sort(proba, axis=1)
    margins = sorted_proba[:, -1] - sorted_proba[:, -2]
    return np.argsort(margins)[-k:]

# –ü—Ä–∏–º–µ—Ä —Å entropy
def select_by_entropy(proba, k):
    """–í—ã–±—Ä–∞—Ç—å k –ø—Ä–∏–º–µ—Ä–æ–≤ —Å –Ω–∞–∏–º–µ–Ω—å—à–µ–π —ç–Ω—Ç—Ä–æ–ø–∏–µ–π"""
    entropy = -np.sum(proba * np.log(proba + 1e-10), axis=1)
    return np.argsort(entropy)[:k]

def evaluate_cotraining(cotrainer, X1_test, X2_test, y_test):
    # –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –∫–∞–∂–¥–æ–π –º–æ–¥–µ–ª–∏
    pred1 = cotrainer.clf1.predict(X1_test)
    pred2 = cotrainer.clf2.predict(X2_test)
    pred_combined = cotrainer.predict(X1_test, X2_test)
    
    # –ú–µ—Ç—Ä–∏–∫–∏
    from sklearn.metrics import accuracy_score
    acc1 = accuracy_score(y_test, pred1)
    acc2 = accuracy_score(y_test, pred2)
    acc_comb = accuracy_score(y_test, pred_combined)
    
    # –°–æ–≥–ª–∞—Å–æ–≤–∞–Ω–Ω–æ—Å—Ç—å
    agreement = np.mean(pred1 == pred2)
    
    return {
        'acc_view1': acc1,
        'acc_view2': acc2,
        'acc_combined': acc_comb,
        'agreement': agreement
    }
