üî∑ 6. Viterbi –∞–ª–≥–æ—Ä–∏—Ç–º

–¶–µ–ª—å: –Ω–∞–π—Ç–∏ –Ω–∞–∏–±–æ–ª–µ–µ –≤–µ—Ä–æ—è—Ç–Ω—É—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Å–æ—Å—Ç–æ—è–Ω–∏–π

# –ü—Å–µ–≤–¥–æ–∫–æ–¥
def viterbi(obs, A, B, pi):
    T = len(obs)
    N = len(pi)
    
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
    viterbi = np.zeros((N, T))
    backpointer = np.zeros((N, T))
    
    viterbi[:, 0] = pi * B[:, obs[0]]
    
    # –†–µ–∫—É—Ä—Å–∏—è
    for t in range(1, T):
        for s in range(N):
            prob = viterbi[:, t-1] * A[:, s]
            backpointer[s, t] = np.argmax(prob)
            viterbi[s, t] = np.max(prob) * B[s, obs[t]]
    
    # Backtracking
    best_path = []
    state = np.argmax(viterbi[:, -1])
    for t in range(T-1, -1, -1):
        best_path.insert(0, state)
        state = int(backpointer[state, t])
    
    return best_path
