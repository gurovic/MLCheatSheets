from sklearn.ensemble import GradientBoostingClassifier
import pandas as pd

# –°–æ–∑–¥–∞–Ω–∏–µ —Ç–∞—Ä–≥–µ—Ç–∞: –±—É–¥–µ—Ç –ª–∏ —Ü–µ–Ω–∞ –≤—ã—à–µ —á–µ—Ä–µ–∑ N –ø–µ—Ä–∏–æ–¥–æ–≤
horizon = 5  # 5 –ø–µ—Ä–∏–æ–¥–æ–≤ –≤–ø–µ—Ä–µ–¥
df['target'] = (df['close'].shift(-horizon) > df['close']).astype(int)

# –ü—Ä–∏–∑–Ω–∞–∫–∏
features = ['returns', 'RSI', 'MACD', 'volume_ratio', 'ATR']
X = df[features].iloc[:-horizon]
y = df['target'].iloc[:-horizon]

# Walk-forward –≤–∞–ª–∏–¥–∞—Ü–∏—è
train_size = int(len(X) * 0.8)
X_train, X_test = X[:train_size], X[train_size:]
y_train, y_test = y[:train_size], y[train_size:]

# Gradient Boosting
gb = GradientBoostingClassifier(n_estimators=100, max_depth=3)
gb.fit(X_train, y_train)

# –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π
proba = gb.predict_proba(X_test)[:, 1]
print(f"Direction Accuracy: {gb.score(X_test, y_test):.3f}")

import numpy as np

def kelly_criterion(win_rate, avg_win, avg_loss):
    """–û–ø—Ç–∏–º–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä –ø–æ–∑–∏—Ü–∏–∏ –ø–æ Kelly"""
    return win_rate - ((1 - win_rate) / (avg_win / avg_loss))

def position_size_volatility(capital, risk_per_trade, volatility):
    """Position sizing –Ω–∞ –æ—Å–Ω–æ–≤–µ –≤–æ–ª–∞—Ç–∏–ª—å–Ω–æ—Å—Ç–∏"""
    risk_amount = capital * risk_per_trade
    shares = risk_amount / (2 * volatility)  # 2 sigma stop
    return int(shares)

  • –ü—Ä–∏–∑–Ω–∞–∫–∏: returns, RSI, MACD, volume –∑–∞ —Ä–∞–∑–Ω—ã–µ –ø–µ—Ä–∏–æ–¥—ã

  • –†–∞–∑–º–µ—Ä –ø–æ–∑–∏—Ü–∏–∏: –Ω–∞ –æ—Å–Ω–æ–≤–µ —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç–∏ –º–æ–¥–µ–ª–∏ (predict_proba)
  • –†–∏—Å–∫-–º–µ–Ω–µ–¥–∂–º–µ–Ω—Ç: stop-loss –Ω–∞ 2%, take-profit –Ω–∞ 4%

üí° –ù–∞—á–Ω–∏—Ç–µ —Å –ø—Ä–æ—Å—Ç–æ–π —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏ –∏ –ø–æ—Å—Ç–µ–ø–µ–Ω–Ω–æ —É—Å–ª–æ–∂–Ω—è–π—Ç–µ, —Ç–µ—Å—Ç–∏—Ä—É—è –∫–∞–∂–¥–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ 
import pandas as pd
import xgboost as xgb
from sklearn.model_selection import TimeSeriesSplit

# –ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö –∏ feature engineering
df = pd.read_csv('btc_usdt_1h.csv')
df['returns'] = df['close'].pct_change()
df['momentum_5'] = df['close'] / df['close'].shift(5) - 1
df['momentum_10'] = df['close'] / df['close'].shift(10) - 1
df['RSI'] = talib.RSI(df['close'])
df['volume_ratio'] = df['volume'] / df['volume'].rolling(20).mean()

# –¢–∞—Ä–≥–µ—Ç: –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —á–µ—Ä–µ–∑ 3 –ø–µ—Ä–∏–æ–¥–∞
df['target'] = (df['close'].shift(-3) > df['close']).astype(int)

# –ü—Ä–∏–∑–Ω–∞–∫–∏ –∏ —Ç–∞—Ä–≥–µ—Ç
features = ['returns', 'momentum_5', 'momentum_10', 'RSI', 'volume_ratio']
X = df[features].fillna(0).iloc[:-3]
y = df['target'].iloc[:-3]

# Walk-forward –≤–∞–ª–∏–¥–∞—Ü–∏—è
tscv = TimeSeriesSplit(n_splits=5)
for train_idx, test_idx in tscv.split(X):
    X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
    y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]
    
    # XGBoost
    model = xgb.XGBClassifier(n_estimators=100, max_depth=3)
    model.fit(X_train, y_train)
    
    # –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è
    proba = model.predict_proba(X_test)[:, 1]
    
    # –¢–æ—Ä–≥–æ–≤—ã–µ —Å–∏–≥–Ω–∞–ª—ã (—Ç–æ–ª—å–∫–æ —Å –≤—ã—Å–æ–∫–æ–π —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å—é)
    signals = pd.Series(0, index=X_test.index)
    signals[proba > 0.6] = 1  # Buy
    signals[proba < 0.4] = -1  # Sell
    
    print(f"Accuracy: {model.score(X_test, y_test):.3f}")
