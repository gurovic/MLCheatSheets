import torch

def vanilla_saliency(model, image, target_class):
    image.requires_grad = True
    
    # Forward pass
    output = model(image)
    
    # Backward –¥–ª—è target –∫–ª–∞—Å—Å–∞
    model.zero_grad()
    output[0, target_class].backward()
    
    # –ì—Ä–∞–¥–∏–µ–Ω—Ç—ã = saliency
    saliency = image.grad.data.abs().max(dim=1)[0]
    
    return saliency

def smooth_grad(model, image, target_class, n_samples=50, noise_level=0.15):
    saliency_maps = []
    
    for _ in range(n_samples):
        # –î–æ–±–∞–≤–ª—è–µ–º –≥–∞—É—Å—Å–æ–≤—Å–∫–∏–π —à—É–º
        noise = torch.randn_like(image) * noise_level
        noisy_image = image + noise
        noisy_image.requires_grad = True
        
        # –í—ã—á–∏—Å–ª—è–µ–º saliency
        output = model(noisy_image)
        model.zero_grad()
        output[0, target_class].backward()
        
        saliency = noisy_image.grad.data.abs()
        saliency_maps.append(saliency)
    
    # –£—Å—Ä–µ–¥–Ω—è–µ–º
    smooth_saliency = torch.stack(saliency_maps).mean(dim=0)
    return smooth_saliency.max(dim=1)[0]

class GuidedReLU(torch.autograd.Function):
    @staticmethod
    def forward(ctx, input):
        ctx.save_for_backward(input)
        return input.clamp(min=0)
    
    @staticmethod
    def backward(ctx, grad_output):
        input, = ctx.saved_tensors
        # –û–±–∞ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å > 0
        grad_input = grad_output.clone()
        grad_input[input < 0] = 0
        grad_input[grad_output < 0] = 0
        return grad_input

import matplotlib.pyplot as plt
import numpy as np

def visualize_saliency(image, saliency, alpha=0.5):
    # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è saliency –∫ [0, 1]
    saliency = (saliency - saliency.min()) / (saliency.max() - saliency.min())
    
    # –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ colormap
    heatmap = plt.cm.jet(saliency.cpu().numpy())[..., :3]
    
    # –ù–∞–ª–æ–∂–µ–Ω–∏–µ –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
    image_np = image.permute(1, 2, 0).cpu().numpy()
    overlay = alpha * heatmap + (1 - alpha) * image_np
    
    plt.figure(figsize=(15, 5))
    plt.subplot(131)
    plt.imshow(image_np)
    plt.title("Original")
    plt.subplot(132)
    plt.imshow(saliency.cpu(), cmap="hot")
    plt.title("Saliency Map")
    plt.subplot(133)
    plt.imshow(overlay)
    plt.title("Overlay")
    plt.show()
