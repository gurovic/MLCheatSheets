# –ó–Ω–∞—á–µ–Ω–∏—è: [-1, 1], —á–µ–º –±–ª–∏–∂–µ –∫ 1, —Ç–µ–º –ª—É—á—à–µ

# Silhouette –¥–ª—è –∫–∞–∂–¥–æ–π —Ç–æ—á–∫–∏
sample_scores = silhouette_samples(X_scaled, labels)

# –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
import numpy as np
for i in range(kmeans.n_clusters):
    cluster_scores = sample_scores[labels == i]
    cluster_scores.sort()
    
    y_lower = i * len(cluster_scores)
    y_upper = y_lower + len(cluster_scores)
    
    plt.fill_betweenx(
        np.arange(y_lower, y_upper),
        0, cluster_scores,
        alpha=0.7
    )

plt.axvline(x=score, color="red", linestyle="--")
plt.xlabel("Silhouette coefficient")
plt.show()
