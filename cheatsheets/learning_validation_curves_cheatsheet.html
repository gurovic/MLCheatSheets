    param_name='alpha',  # –ø–∞—Ä–∞–º–µ—Ç—Ä –¥–ª—è –Ω–∞—Å—Ç—Ä–æ–π–∫–∏

plt.ylabel('Score')
plt.legend()
plt.title('Validation Curve')
plt.grid(True)
plt.show()
