@qml.qnode(dev)
def circuit(weights, x):

    qml.RY(x[1], wires=1)
    qml.CNOT(wires=[0, 1])
    qml.RX(weights[0], wires=0)
    qml.RY(weights[1], wires=1)
    return qml.expval(qml.PauliZ(0))
