import torch

def integrated_gradients(model, image, target_class, baseline=None, steps=50):
    if baseline is None:
        baseline = torch.zeros_like(image)
    
    # –õ–∏–Ω–µ–π–Ω–∞—è –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è –æ—Ç baseline –∫ image
    alphas = torch.linspace(0, 1, steps)
    
    # –ù–∞–∫–æ–ø–ª–µ–Ω–∏–µ –≥—Ä–∞–¥–∏–µ–Ω—Ç–æ–≤
    integrated_grads = torch.zeros_like(image)
    
    for alpha in alphas:
        # –ò–Ω—Ç–µ—Ä–ø–æ–ª–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
        interpolated = baseline + alpha * (image - baseline)
        interpolated.requires_grad = True
        
        # Forward + backward
        output = model(interpolated)
        model.zero_grad()
        output[0, target_class].backward()
        
        # –î–æ–±–∞–≤–ª—è–µ–º –≥—Ä–∞–¥–∏–µ–Ω—Ç
        integrated_grads += interpolated.grad.data
    
    # –£—Å—Ä–µ–¥–Ω—è–µ–º –∏ —É–º–Ω–æ–∂–∞–µ–º –Ω–∞ (x - x')
    integrated_grads = integrated_grads / steps
    attributions = (image - baseline) * integrated_grads
    
    return attributions

# Batch processing –¥–ª—è —É—Å–∫–æ—Ä–µ–Ω–∏—è
def integrated_gradients_batch(model, image, target_class, steps=50, batch_size=10):
    baseline = torch.zeros_like(image)
    alphas = torch.linspace(0, 1, steps)
    
    all_grads = []
    
    for i in range(0, steps, batch_size):
        batch_alphas = alphas[i:i+batch_size]
        # –°–æ–∑–¥–∞—ë–º batch –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏–π
        interpolated_batch = torch.stack([
            baseline + alpha * (image - baseline) 
            for alpha in batch_alphas
        ])
        interpolated_batch.requires_grad = True
        
        # Batch forward
        outputs = model(interpolated_batch)
        
        # Backward –¥–ª—è target –∫–ª–∞—Å—Å–∞
        model.zero_grad()
        target_outputs = outputs[:, target_class].sum()
        target_outputs.backward()
        
        all_grads.append(interpolated_batch.grad.data)
    
    # –£—Å—Ä–µ–¥–Ω–µ–Ω–∏–µ
    integrated_grads = torch.cat(all_grads).mean(dim=0)
    attributions = (image - baseline) * integrated_grads
    
    return attributions
