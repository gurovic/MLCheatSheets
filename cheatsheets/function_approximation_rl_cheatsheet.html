import numpy as np

def tile_coding(state, num_tilings=8, tiles_per_dim=8):
    features = np.zeros(num_tilings * tiles_per_dim**2)
    for i in range(num_tilings):
        offset = i / num_tilings
        idx = int((state + offset) * tiles_per_dim)
        features[i * tiles_per_dim + idx] = 1
    return features
