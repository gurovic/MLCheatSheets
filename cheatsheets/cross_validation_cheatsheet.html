    model, X, y,
    cv=skf,
    scoring='f1_weighted'
)

print(f"Mean F1: {scores.mean():.3f}")

    cv=5,
    scoring=scoring,
    return_train_score=True
)

    test_score = cv_results[f'test_{metric}']
    print(f"{metric}: {test_score.mean():.3f} (+/- {test_score.std():.3f})")

kf = KFold(n_splits=5, shuffle=True, random_state=42)

for fold, (train_idx, val_idx) in enumerate(kf.split(X), 1):
    print(f"\nFold {fold}")
    
    X_train, X_val = X[train_idx], X[val_idx]
    y_train, y_val = y[train_idx], y[val_idx]
    
    model.fit(X_train, y_train)
    train_score = model.score(X_train, y_train)
    val_score = model.score(X_val, y_val)
    
    print(f"Train: {train_score:.3f}, Val: {val_score:.3f}")

üî∑ 8. GridSearchCV —Å CV

from sklearn.model_selection import GridSearchCV

    'n_estimators': [100, 200, 300],
    'max_depth': [10, 20, 30],
    'min_samples_split': [2, 5, 10]
}

grid_search = GridSearchCV(
    RandomForestClassifier(random_state=42),
    param_grid,
    cv=5,  # –í–Ω—É—Ç—Ä–µ–Ω–Ω—è—è –∫—Ä–æ—Å—Å-–≤–∞–ª–∏–¥–∞—Ü–∏—è
    scoring='f1_weighted',
    n_jobs=-1,
    verbose=1
)

grid_search.fit(X, y)

print("Best params:", grid_search.best_params_)
print("Best CV score:", grid_search.best_score_)
