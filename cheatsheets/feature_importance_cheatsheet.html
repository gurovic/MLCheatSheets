üî∑ 6. Drop Column Importance

–£–¥–∞–ª–µ–Ω–∏–µ –ø—Ä–∏–∑–Ω–∞–∫–∞ –∏ –∏–∑–º–µ—Ä–µ–Ω–∏–µ –ø–∞–¥–µ–Ω–∏—è –∫–∞—á–µ—Å—Ç–≤–∞:

from sklearn.model_selection import cross_val_score

def drop_col_importance(model, X, y, cv=5):
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ –≤–∞–∂–Ω–æ—Å—Ç–∏ —á–µ—Ä–µ–∑ —É–¥–∞–ª–µ–Ω–∏–µ —Å—Ç–æ–ª–±—Ü–æ–≤"""
    # –ë–∞–∑–æ–≤–∞—è –æ—Ü–µ–Ω–∫–∞
    base_score = cross_val_score(
        model, X, y, cv=cv, 
        scoring='accuracy'
    ).mean()
    
    importances = []
    for col_idx in range(X.shape[1]):
        # –£–¥–∞–ª—è–µ–º —Å—Ç–æ–ª–±–µ—Ü
        X_dropped = np.delete(X, col_idx, axis=1)
        
        # –û—Ü–µ–Ω–∫–∞ –±–µ–∑ –ø—Ä–∏–∑–Ω–∞–∫–∞
        score = cross_val_score(
            model, X_dropped, y, 
            cv=cv, scoring='accuracy'
        ).mean()
        
        # –†–∞–∑–Ω–∏—Ü–∞ = –≤–∞–∂–Ω–æ—Å—Ç—å
        importance = base_score - score
        importances.append(importance)
    
    return np.array(importances)

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
model = RandomForestClassifier(n_estimators=50)
importances = drop_col_importance(model, X, y)

# –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
indices = np.argsort(importances)[::-1]
plt.bar(range(len(importances)), 
        importances[indices])
plt.xticks(range(len(importances)),
           [feature_names[i] for i in indices],
           rotation=45)
plt.title('Drop Column Importance')
plt.tight_layout()
plt.show()

# –ü—Ä–æ–≤–µ—Ä–∫–∞ VIF (Variance Inflation Factor)
from statsmodels.stats.outliers_influence import variance_inflation_factor

def calculate_vif(df):
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ VIF –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –ø—Ä–∏–∑–Ω–∞–∫–∞"""

    vif_data["feature"] = df.columns
    vif_data["VIF"] = [
        variance_inflation_factor(df.values, i) 
        for i in range(len(df.columns))
    ]
    return vif_data.sort_values('VIF', ascending=False)
