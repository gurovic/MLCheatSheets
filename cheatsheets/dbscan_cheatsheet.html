from kneed import KneeLocator

# –ù–∞–π—Ç–∏ —Ç–æ—á–∫—É –ø–µ—Ä–µ–≥–∏–±–∞ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å KneeLocator

    range(len(k_distances)), 
    k_distances,
    curve='convex',
    direction='increasing'
)

# –û–ø—Ç–∏–º–∞–ª—å–Ω—ã–π eps
optimal_eps = k_distances[kneedle.knee]
print(f"–†–µ–∫–æ–º–µ–Ω–¥—É–µ–º—ã–π eps: {optimal_eps:.3f}")

üî∑ 14. Grid Search –¥–ª—è –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤

from sklearn.metrics import silhouette_score

# –ü–æ–¥–±–æ—Ä –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤
best_score = -1
best_params = {}

eps_range = np.linspace(0.1, 2.0, 20)
min_samples_range = [3, 5, 7, 10]

for eps in eps_range:
    for min_samples in min_samples_range:
        dbscan = DBSCAN(eps=eps, min_samples=min_samples)
        labels = dbscan.fit_predict(X_scaled)
        
        # –ü—Ä–æ–ø—É—Å—Ç–∏—Ç—å, –µ—Å–ª–∏ –≤—Å–µ —Ç–æ—á–∫–∏ - —à—É–º –∏–ª–∏ –æ–¥–∏–Ω –∫–ª–∞—Å—Ç–µ—Ä
        n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
        if n_clusters < 2:
            continue
            
        # –ú–µ—Ç—Ä–∏–∫–∞ —Ç–æ–ª—å–∫–æ –¥–ª—è –∫–ª–∞—Å—Ç–µ—Ä–æ–≤
        mask = labels != -1
        if mask.sum() < 2:
            continue
            
        score = silhouette_score(X_scaled[mask], labels[mask])
        
        if score > best_score:
            best_score = score
            best_params = {'eps': eps, 'min_samples': min_samples}

print(f"Best params: {best_params}")
print(f"Best score: {best_score:.3f}")
