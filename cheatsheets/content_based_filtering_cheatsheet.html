movies = pd.DataFrame({

    'genres': ['Action Sci-Fi', 'Drama Romance', 
               'Action Sci-Fi Adventure', 'Action Comedy'],
    'director': ['Cameron', 'Cameron', 'Cameron', 'Cameron'],
    'year': [1984, 1997, 2009, 1994]

def get_recommendations(title, top_n=3):
    # –ù–∞–π—Ç–∏ –∏–Ω–¥–µ–∫—Å —Ñ–∏–ª—å–º–∞
    idx = movies[movies['title'] == title].index[0]
    
    # –ü–æ–ª—É—á–∏—Ç—å –æ—Ü–µ–Ω–∫–∏ —Å—Ö–æ–∂–µ—Å—Ç–∏
    sim_scores = list(enumerate(cosine_sim[idx]))
    
    # –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –ø–æ —Å—Ö–æ–∂–µ—Å—Ç–∏
    sim_scores = sorted(sim_scores, 
                       key=lambda x: x[1], 
                       reverse=True)
    
    # –ü–æ–ª—É—á–∏—Ç—å top N (–∏—Å–∫–ª—é—á–∞—è —Å–∞–º —Ñ–∏–ª—å–º)
    sim_scores = sim_scores[1:top_n+1]
    
    # –ò–Ω–¥–µ–∫—Å—ã –ø–æ—Ö–æ–∂–∏—Ö —Ñ–∏–ª—å–º–æ–≤
    movie_indices = [i[0] for i in sim_scores]
    
    return movies['title'].iloc[movie_indices]

# –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–∏

class ContentBasedRecommender:
    def __init__(self, items_df, feature_cols):
        self.items = items_df
        self.feature_cols = feature_cols
        self.tfidf = TfidfVectorizer()
        self.item_profiles = None
        
    def fit(self):
        # –û–±—ä–µ–¥–∏–Ω–∏—Ç—å –ø—Ä–∏–∑–Ω–∞–∫–∏ –≤ —Ç–µ–∫—Å—Ç
        features_text = self.items[self.feature_cols].apply(
            lambda x: ' '.join(x.astype(str)), axis=1
        )
        
        # –°–æ–∑–¥–∞—Ç—å –º–∞—Ç—Ä–∏—Ü—É –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
        self.item_profiles = self.tfidf.fit_transform(features_text)
        return self
    
    def build_user_profile(self, user_items, ratings=None):
        # –ü—Ä–æ—Ñ–∏–ª—å –Ω–∞ –æ—Å–Ω–æ–≤–µ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏–π
        if ratings is None:
            # –ü—Ä–æ—Å—Ç–æ–µ —É—Å—Ä–µ–¥–Ω–µ–Ω–∏–µ
            return self.item_profiles[user_items].mean(axis=0)
        else:
            # –í–∑–≤–µ—à–µ–Ω–Ω–æ–µ –ø–æ —Ä–µ–π—Ç–∏–Ω–≥–∞–º
            weights = np.array(ratings) / sum(ratings)
            return (self.item_profiles[user_items].T * weights).T.sum(axis=0)
    
    def recommend(self, user_profile, top_n=10, exclude_items=None):
        # –í—ã—á–∏—Å–ª–∏—Ç—å —Å—Ö–æ–¥—Å—Ç–≤–æ
        similarities = cosine_similarity(user_profile, self.item_profiles)
        
        # –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å
        indices = np.argsort(similarities[0])[::-1]
        
        # –ò—Å–∫–ª—é—á–∏—Ç—å —É–∂–µ –ø—Ä–æ—Å–º–æ—Ç—Ä–µ–Ω–Ω—ã–µ
        if exclude_items:
            indices = [i for i in indices if i not in exclude_items]
        
        return indices[:top_n]

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
recommender = ContentBasedRecommender(movies, ['genres', 'director'])
recommender.fit()

user_items = [0, 2]  # –ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é –ø–æ–Ω—Ä–∞–≤–∏–ª–∏—Å—å —Ñ–∏–ª—å–º—ã 0 –∏ 2
user_profile = recommender.build_user_profile(user_items)
recommendations = recommender.recommend(user_profile, top_n=5, 
                                       exclude_items=user_items)

    text = text.lower()
    
    # –£–¥–∞–ª–∏—Ç—å –ø—É–Ω–∫—Ç—É–∞—Ü–∏—é
    text = ''.join([c for c in text if c.isalnum() or c.isspace()])

    return ' '.join(tokens)

def diversify_recommendations(items, similarities, diversity_weight=0.3):
    """–ú–∞–∫—Å–∏–º–∏–∑–∏—Ä–æ–≤–∞—Ç—å —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ—Å—Ç—å –∏ —Ä–∞–∑–Ω–æ–æ–±—Ä–∞–∑–∏–µ"""
    selected = []
    candidates = list(range(len(items)))
    
    # –í—ã–±—Ä–∞—Ç—å —Å–∞–º—ã–π —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–π
    best_idx = np.argmax(similarities)
    selected.append(best_idx)
    candidates.remove(best_idx)
    
    # –ò—Ç–µ—Ä–∞—Ç–∏–≤–Ω–æ –¥–æ–±–∞–≤–ª—è—Ç—å —Ä–∞–∑–Ω–æ–æ–±—Ä–∞–∑–Ω—ã–µ
    while len(selected) < top_n and candidates:
        scores = []
        for c in candidates:
            # –†–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ—Å—Ç—å
            relevance = similarities[c]
            
            # –†–∞–∑–Ω–æ–æ–±—Ä–∞–∑–∏–µ (–º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ —Å—Ö–æ–¥—Å—Ç–≤–æ —Å –≤—ã–±—Ä–∞–Ω–Ω—ã–º–∏)
            diversity = min([1 - cosine_similarity(
                item_profiles[c], item_profiles[s]
            )[0][0] for s in selected])
            
            # –ö–æ–º–±–∏–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–π score
            score = relevance * (1 - diversity_weight) + diversity * diversity_weight
            scores.append((c, score))
        
        # –í—ã–±—Ä–∞—Ç—å –ª—É—á—à–∏–π
        best = max(scores, key=lambda x: x[1])
        selected.append(best[0])
        candidates.remove(best[0])
    
    return selected
