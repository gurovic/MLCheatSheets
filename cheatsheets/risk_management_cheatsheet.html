  • Model performance: AUC, Gini –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç
  • Calibration: —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –ø—Ä–æ–≥–Ω–æ–∑–æ–≤ —Ä–µ–∞–ª—å–Ω–æ—Å—Ç–∏
def calculate_psi(expected, actual, bins=10):
    """Population Stability Index"""
    exp_pct = pd.cut(expected, bins=bins).value_counts() / len(expected)
    act_pct = pd.cut(actual, bins=bins).value_counts() / len(actual)
    
    psi = ((act_pct - exp_pct) * np.log(act_pct / exp_pct)).sum()
    return psi
