import pyro
import pyro.distributions as dist

def model(data):
    # Priors
    mu = pyro.sample('mu', dist.Normal(0, 10))
    sigma = pyro.sample('sigma', dist.HalfNormal(10))
    
    # Likelihood
    with pyro.plate('data', len(data)):
        pyro.sample('obs', dist.Normal(mu, sigma), obs=data)
