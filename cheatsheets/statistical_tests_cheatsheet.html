1. T-test

from scipy import stats

# –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –¥–≤—É—Ö –≤—ã–±–æ—Ä–æ–∫
group1 = [1, 2, 3, 4, 5]
group2 = [2, 3, 4, 5, 6]

    print("–†–∞–∑–ª–∏—á–∏—è –∑–Ω–∞—á–∏–º—ã")

3. ANOVA

from scipy.stats import f_oneway

group1 = [1, 2, 3]
group2 = [4, 5, 6]
group3 = [7, 8, 9]

f_stat, p_value = f_oneway(group1, group2, group3)
print(f"F-statistic: {f_stat:.2f}")
print(f"p-value: {p_value:.4f}")
