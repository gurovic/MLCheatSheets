import numpy as np

class KalmanFilter1D:
    def __init__(self, F=1, H=1, Q=1, R=1, x0=0, P0=1):
        self.F = F  # –ú–∞—Ç—Ä–∏—Ü–∞ –ø–µ—Ä–µ—Ö–æ–¥–∞
        self.H = H  # –ú–∞—Ç—Ä–∏—Ü–∞ –Ω–∞–±–ª—é–¥–µ–Ω–∏—è
        self.Q = Q  # –®—É–º –ø—Ä–æ—Ü–µ—Å—Å–∞
        self.R = R  # –®—É–º –∏–∑–º–µ—Ä–µ–Ω–∏–π
        self.x = x0  # –ù–∞—á–∞–ª—å–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ
        self.P = P0  # –ù–∞—á–∞–ª—å–Ω–∞—è –∫–æ–≤–∞—Ä–∏–∞—Ü–∏—è
    
    def predict(self, u=0):
        # –ü—Ä–æ–≥–Ω–æ–∑ —Å–æ—Å—Ç–æ—è–Ω–∏—è
        self.x = self.F * self.x + u
        self.P = self.F * self.P * self.F + self.Q
        return self.x
    
    def update(self, z):
        # –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —É—Å–∏–ª–µ–Ω–∏—è –ö–∞–ª–º–∞–Ω–∞
        K = self.P * self.H / (self.H * self.P * self.H + self.R)
        
        # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –æ—Ü–µ–Ω–∫–∏
        self.x = self.x + K * (z - self.H * self.x)
        
        # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –∫–æ–≤–∞—Ä–∏–∞—Ü–∏–∏
        self.P = (1 - K * self.H) * self.P
        
        return self.x

    estimate = kf.update(z)
    estimates.append(estimate)

import numpy as np

class KalmanFilter2D:
    def __init__(self, dt=1.0):
        # –°–æ—Å—Ç–æ—è–Ω–∏–µ: [x, y, vx, vy]
        self.dt = dt
        
        # –ú–∞—Ç—Ä–∏—Ü–∞ –ø–µ—Ä–µ—Ö–æ–¥–∞ —Å–æ—Å—Ç–æ—è–Ω–∏—è
        self.F = np.array([
            [1, 0, dt, 0],
            [0, 1, 0, dt],
            [0, 0, 1, 0],
            [0, 0, 0, 1]
        ])
        
        # –ú–∞—Ç—Ä–∏—Ü–∞ –Ω–∞–±–ª—é–¥–µ–Ω–∏—è (–∏–∑–º–µ—Ä—è–µ–º —Ç–æ–ª—å–∫–æ –ø–æ–∑–∏—Ü–∏—é)
        self.H = np.array([
            [1, 0, 0, 0],
            [0, 1, 0, 0]
        ])
        
        # –ö–æ–≤–∞—Ä–∏–∞—Ü–∏—è —à—É–º–∞ –ø—Ä–æ—Ü–µ—Å—Å–∞
        self.Q = np.eye(4) * 0.1
        
        # –ö–æ–≤–∞—Ä–∏–∞—Ü–∏—è —à—É–º–∞ –∏–∑–º–µ—Ä–µ–Ω–∏–π
        self.R = np.eye(2) * 1.0
        
        # –ù–∞—á–∞–ª—å–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ
        self.x = np.zeros(4)
        self.P = np.eye(4) * 100
    
    def predict(self):
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q
        return self.x[:2]
    
    def update(self, z):
        y = z - self.H @ self.x
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        self.x = self.x + K @ y
        self.P = (np.eye(4) - K @ self.H) @ self.P
        
        return self.x[:2]

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

measurements = [(1, 1), (2, 2), (3, 3), (4, 4)]

for z in measurements:

    position = kf.update(np.array(z))
    print(f"–û—Ü–µ–Ω–∫–∞ –ø–æ–∑–∏—Ü–∏–∏: {position}")

    # –Ø–∫–æ–±–∏–∞–Ω —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞–±–ª—é–¥–µ–Ω–∏—è

class ParticleFilter:
    def __init__(self, n_particles):
        self.n_particles = n_particles
        self.particles = np.random.randn(n_particles, 4)
        self.weights = np.ones(n_particles) / n_particles
    
    def predict(self, u, noise_std):
        # –ü—Ä–æ–≥–Ω–æ–∑ —Å –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ–º —à—É–º–∞
        self.particles += u + np.random.randn(
            self.n_particles, 4) * noise_std
    
    def update(self, z, measurement_func, noise_std):
        # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Å–æ–≤
        for i in range(self.n_particles):
            predicted_z = measurement_func(self.particles[i])
            self.weights[i] *= self._gaussian_likelihood(
                z, predicted_z, noise_std)
        
        self.weights += 1.e-300  # –ò–∑–±–µ–∂–∞—Ç—å –¥–µ–ª–µ–Ω–∏—è –Ω–∞ –Ω–æ–ª—å
        self.weights /= sum(self.weights)
    
    def resample(self):
        # –†–µ—Å–µ–º–ø–ª–∏–Ω–≥ —á–∞—Å—Ç–∏—Ü
        indices = np.random.choice(
            self.n_particles, 
            self.n_particles, 
            p=self.weights
        )
        self.particles = self.particles[indices]
        self.weights.fill(1.0 / self.n_particles)
    
    def estimate(self):
        return np.average(self.particles, weights=self.weights, axis=0)
