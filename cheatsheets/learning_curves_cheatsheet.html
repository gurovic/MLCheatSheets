def diagnose_model(model, X, y):
    """–ü–æ–ª–Ω–∞—è –¥–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–∞ –º–æ–¥–µ–ª–∏"""
    
    # Learning curve
    train_sizes, train_scores, val_scores = learning_curve(
        model, X, y,
        train_sizes=np.linspace(0.1, 1.0, 10),
        cv=5, n_jobs=-1
    )
    
    train_mean = train_scores.mean(axis=1)
    val_mean = val_scores.mean(axis=1)
    
    # –§–∏–Ω–∞–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è
    final_train = train_mean[-1]
    final_val = val_mean[-1]
    gap = final_train - final_val
    
    print(f"Train score: {final_train:.3f}")
    print(f"Val score: {final_val:.3f}")
    print(f"Gap: {gap:.3f}")
    
    # –î–∏–∞–≥–Ω–æ–∑
    if gap > 0.1 and final_train > 0.9:
        print("‚ùå –ü–ï–†–ï–û–ë–£–ß–ï–ù–ò–ï")

    elif final_train < 0.8 and gap < 0.05:
        print("‚ùå –ù–ï–î–û–û–ë–£–ß–ï–ù–ò–ï")
        print("‚Üí –£—Å–ª–æ–∂–Ω–∏ –º–æ–¥–µ–ª—å, –¥–æ–±–∞–≤—å –ø—Ä–∏–∑–Ω–∞–∫–∏")
    elif gap < 0.05 and final_val > 0.85:
        print("‚úÖ –•–û–†–û–®–ê–Ø –ú–û–î–ï–õ–¨")

        print("‚ö†Ô∏è –ù–£–ñ–ù–ê –ù–ê–°–¢–†–û–ô–ö–ê")

    plt.plot(train_sizes, val_mean, 'o-', label='Val')
    plt.xlabel('Training Size')
    plt.ylabel('Score')
    plt.title('Learning Curves')
    plt.legend()
    plt.grid(True)

üî∑ 15. Bootstrap Confidence Intervals

from sklearn.utils import resample

def bootstrap_learning_curve(model, X, y, n_bootstrap=100):
    """Learning curve —Å –¥–æ–≤–µ—Ä–∏—Ç–µ–ª—å–Ω—ã–º–∏ –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞–º–∏"""
    
    results = []
    
    for _ in range(n_bootstrap):
        # Resample
        X_boot, y_boot = resample(X, y)
        
        # Learning curve
        _, train_scores, val_scores = learning_curve(
            model, X_boot, y_boot,
            train_sizes=np.linspace(0.1, 1.0, 10),
            cv=3
        )
        
        results.append({
            'train': train_scores.mean(axis=1),
            'val': val_scores.mean(axis=1)
        })
    
    # Percentiles
    train_lower = np.percentile([r['train'] for r in results], 2.5, axis=0)
    train_upper = np.percentile([r['train'] for r in results], 97.5, axis=0)
    val_lower = np.percentile([r['val'] for r in results], 2.5, axis=0)
    val_upper = np.percentile([r['val'] for r in results], 97.5, axis=0)
    
    return train_lower, train_upper, val_lower, val_upper

from sklearn.linear_model import SGDClassifier

def incremental_learning_curve(X, y, batch_size=1000):
    """–ò–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª—å–Ω–æ–µ –æ–±—É—á–µ–Ω–∏–µ"""
    
    model = SGDClassifier()
    train_scores = []
    val_scores = []
    sizes = []
    
    X_train, X_val, y_train, y_val = train_test_split(
        X, y, test_size=0.2
    )
    
    n_batches = len(X_train) // batch_size
    
    for i in range(1, n_batches + 1):
        # –û–±—É—á–µ–Ω–∏–µ –Ω–∞ i –±–∞—Ç—á–∞—Ö
        X_batch = X_train[:i*batch_size]
        y_batch = y_train[:i*batch_size]
        
        model.partial_fit(X_batch, y_batch, classes=np.unique(y))
        
        # –û—Ü–µ–Ω–∫–∞
        train_score = model.score(X_batch, y_batch)
        val_score = model.score(X_val, y_val)
        
        train_scores.append(train_score)
        val_scores.append(val_score)
        sizes.append(i * batch_size)
    
    plt.plot(sizes, train_scores, label='Train')
    plt.plot(sizes, val_scores, label='Val')
    plt.xlabel('Training Size')
    plt.ylabel('Accuracy')
    plt.legend()
    plt.grid(True)
    plt.show()
