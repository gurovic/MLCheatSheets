def one_way_anova_manual(groups):
    """–†—É—á–Ω–æ–π —Ä–∞—Å—á–µ—Ç One-Way ANOVA"""
    k = len(groups)  # —á–∏—Å–ª–æ –≥—Ä—É–ø–ø
    n = sum(len(g) for g in groups)  # –æ–±—â–µ–µ —á–∏—Å–ª–æ –Ω–∞–±–ª—é–¥–µ–Ω–∏–π
    
    # –û–±—â–µ–µ —Å—Ä–µ–¥–Ω–µ–µ
    grand_mean = np.mean([x for g in groups for x in g])
    
    # SST (Total Sum of Squares)
    sst = sum((x - grand_mean)**2 
             for g in groups for x in g)
    
    # SSB (Between-group Sum of Squares)
    ssb = sum(len(g) * (np.mean(g) - grand_mean)**2 
             for g in groups)
    
    # SSW (Within-group Sum of Squares)
    ssw = sum((x - np.mean(g))**2 
             for g in groups for x in g)
    
    # Degrees of freedom
    df_between = k - 1
    df_within = n - k
    
    # Mean squares
    msb = ssb / df_between
    msw = ssw / df_within
    
    # F-statistic
    f_stat = msb / msw
    
    # p-value
    p_value = 1 - stats.f.cdf(f_stat, df_between, df_within)
    
    return {
        'F': f_stat,
        'p_value': p_value,
        'SSB': ssb,
        'SSW': ssw,
        'SST': sst,
        'df_between': df_between,
        'df_within': df_within
    }

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
groups = [group1, group2, group3]
results = one_way_anova_manual(groups)
print(f"F = {results['F']:.4f}, p = {results['p_value']:.4f}")

import pandas as pd

def anova_table(groups):
    results = one_way_anova_manual(groups)
    
    table = pd.DataFrame({
        'Source': ['Between', 'Within', 'Total'],
        'SS': [results['SSB'], results['SSW'], results['SST']],
        'df': [results['df_between'], results['df_within'], 
               results['df_between'] + results['df_within']],
        'MS': [results['SSB']/results['df_between'],
               results['SSW']/results['df_within'], 
               np.nan],
        'F': [results['F'], np.nan, np.nan],
        'p-value': [results['p_value'], np.nan, np.nan]
    })
    
    return table

print(anova_table(groups))

# Bonferroni correction (—Ä—É—á–Ω–æ–π)
from itertools import combinations

def bonferroni_test(groups, alpha=0.05):
    k = len(groups)
    n_comparisons = k * (k - 1) // 2
    adjusted_alpha = alpha / n_comparisons
    
    print(f"Adjusted alpha: {adjusted_alpha:.4f}\n")
    
    for (i, g1), (j, g2) in combinations(enumerate(groups), 2):
        t_stat, p = stats.ttest_ind(g1, g2)
        sig = "‚úì" if p < adjusted_alpha else "‚úó"
        print(f"Group {i} vs {j}: p={p:.4f} {sig}")

bonferroni_test(groups)

def eta_squared(groups):
    """Eta-squared: –¥–æ–ª—è –æ–±—ä—è—Å–Ω–µ–Ω–Ω–æ–π –¥–∏—Å–ø–µ—Ä—Å–∏–∏"""
    results = one_way_anova_manual(groups)
    return results['SSB'] / results['SST']

def omega_squared(groups):
    """Omega-squared: –Ω–µ—Å–º–µ—â–µ–Ω–Ω–∞—è –æ—Ü–µ–Ω–∫–∞"""
    results = one_way_anova_manual(groups)
    k = len(groups)
    n = sum(len(g) for g in groups)
    
    numerator = results['SSB'] - (k - 1) * results['SSW'] / (n - k)
    denominator = results['SST'] + results['SSW'] / (n - k)
    
    return numerator / denominator

eta2 = eta_squared(groups)
omega2 = omega_squared(groups)

print(f"Œ∑¬≤ = {eta2:.4f}")
print(f"œâ¬≤ = {omega2:.4f}")
