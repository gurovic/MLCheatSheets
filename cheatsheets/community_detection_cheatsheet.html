def label_propagation(G, max_iter=100):
    """
    Label Propagation –∞–ª–≥–æ—Ä–∏—Ç–º
    –ë—ã—Å—Ç—Ä—ã–π –∏ –ø—Ä–æ—Å—Ç–æ–π –º–µ—Ç–æ–¥
    """
    import random
    
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —É–Ω–∏–∫–∞–ª—å–Ω—ã–º–∏ –º–µ—Ç–∫–∞–º–∏
    labels = {node: node for node in G.nodes()}
    
    for iteration in range(max_iter):
        # –°–ª—É—á–∞–π–Ω—ã–π –ø–æ—Ä—è–¥–æ–∫ —É–∑–ª–æ–≤
        nodes = list(G.nodes())
        random.shuffle(nodes)
        
        changed = False
        for node in nodes:
            # –ü–æ–¥—Å—á–µ—Ç –º–µ—Ç–æ–∫ —Å–æ—Å–µ–¥–µ–π
            neighbor_labels = [labels[neighbor] 
                             for neighbor in G.neighbors(node)]
            
            if not neighbor_labels:
                continue
            
            # –í—ã–±–æ—Ä —Å–∞–º–æ–π —á–∞—Å—Ç–æ–π –º–µ—Ç–∫–∏
            from collections import Counter
            most_common = Counter(neighbor_labels).most_common(1)[0][0]
            
            if labels[node] != most_common:
                labels[node] = most_common
                changed = True
        
        if not changed:
            break
    
    return labels

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

# –∏–ª–∏ —á–µ—Ä–µ–∑ NetworkX

from sklearn.cluster import SpectralClustering
import numpy as np

# –ú–∞—Ç—Ä–∏—Ü–∞ —Å–º–µ–∂–Ω–æ—Å—Ç–∏
A = nx.to_numpy_array(G)

# –í—Ä—É—á–Ω—É—é
def spectral_clustering_manual(A, k):
    """
    –°–ø–µ–∫—Ç—Ä–∞–ª—å–Ω–∞—è –∫–ª–∞—Å—Ç–µ—Ä–∏–∑–∞—Ü–∏—è
    """
    from scipy.sparse.linalg import eigsh
    
    # –°—Ç–µ–ø–µ–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞
    D = np.diag(A.sum(axis=1))
    
    # –õ–∞–ø–ª–∞—Å–∏–∞–Ω
    L = D - A
    
    # –ù–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—ã–π –ª–∞–ø–ª–∞—Å–∏–∞–Ω
    D_inv_sqrt = np.diag(1.0 / np.sqrt(D.diagonal()))
    L_norm = D_inv_sqrt @ L @ D_inv_sqrt
    
    # k –Ω–∞–∏–º–µ–Ω—å—à–∏—Ö —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã—Ö –≤–µ–∫—Ç–æ—Ä–æ–≤
    eigenvalues, eigenvectors = eigsh(L_norm, k=k, which='SM')
    
    # K-means –Ω–∞ —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã—Ö –≤–µ–∫—Ç–æ—Ä–∞—Ö
    from sklearn.cluster import KMeans
    kmeans = KMeans(n_clusters=k)
    labels = kmeans.fit_predict(eigenvectors)
    
    return labels

from sklearn.metrics import adjusted_rand_score, normalized_mutual_info_score

def evaluate_communities(true_labels, pred_labels):
    """
    –û—Ü–µ–Ω–∫–∞ –∫–∞—á–µ—Å—Ç–≤–∞ –æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏—è —Å–æ–æ–±—â–µ—Å—Ç–≤
    """
    # ARI (Adjusted Rand Index)
    ari = adjusted_rand_score(true_labels, pred_labels)
    
    # NMI (Normalized Mutual Information)
    nmi = normalized_mutual_info_score(true_labels, pred_labels)
    
    print(f"ARI: {ari:.3f}")
    print(f"NMI: {nmi:.3f}")
    
    return ari, nmi

# –í–Ω—É—Ç—Ä–µ–Ω–Ω–∏–µ –º–µ—Ç—Ä–∏–∫–∏

from scipy.cluster.hierarchy import dendrogram, linkage
import matplotlib.pyplot as plt

def hierarchical_communities(G):
    """
    –ò–µ—Ä–∞—Ä—Ö–∏—á–µ—Å–∫–∞—è –∫–ª–∞—Å—Ç–µ—Ä–∏–∑–∞—Ü–∏—è –¥–ª—è –≥—Ä–∞—Ñ–æ–≤
    """
    # –ú–∞—Ç—Ä–∏—Ü–∞ –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –ø—É—Ç–µ–π
    path_length = dict(nx.all_pairs_shortest_path_length(G))
    
    n = len(G.nodes())
    distance_matrix = np.zeros((n, n))
    
    nodes = list(G.nodes())
    for i, u in enumerate(nodes):
        for j, v in enumerate(nodes):
            if u in path_length and v in path_length[u]:
                distance_matrix[i][j] = path_length[u][v]
            else:
                distance_matrix[i][j] = n  # Disconnected
    
    # Hierarchical clustering
    Z = linkage(distance_matrix, method='ward')
    
    # Dendrogram
    plt.figure(figsize=(10, 5))
    dendrogram(Z, labels=nodes)
    plt.title('Hierarchical Community Structure')
    plt.show()
    
    return Z
