class RuleValidator:
    def __init__(self):
        self.rules = []
    
    def add_rule(self, rule_func, message):
        self.rules.append((rule_func, message))
    
    def validate(self, prediction, features):
        violations = []
        for rule, msg in self.rules:
            if not rule(prediction, features):
                violations.append(msg)
        return violations
