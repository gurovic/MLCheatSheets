from statsmodels.stats.contingency_tables import mcnemar

# –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –¥–≤—É—Ö –º–æ–¥–µ–ª–µ–π
y_pred1 = model1.predict(X_test)
y_pred2 = model2.predict(X_test)

# –¢–∞–±–ª–∏—Ü–∞ —Å–æ–ø—Ä—è–∂–µ–Ω–Ω–æ—Å—Ç–∏
# –°—Ç—Ä–æ–∫–∏: model1, –°—Ç–æ–ª–±—Ü—ã: model2
contingency_table = np.array([
    [np.sum((y_pred1 == y_test) & (y_pred2 == y_test)),
     np.sum((y_pred1 == y_test) & (y_pred2 != y_test))],
    [np.sum((y_pred1 != y_test) & (y_pred2 == y_test)),
     np.sum((y_pred1 != y_test) & (y_pred2 != y_test))]
])

# McNemar —Ç–µ—Å—Ç
result = mcnemar(contingency_table, exact=True)
print(f"McNemar p-value: {result.pvalue:.4f}")

if result.pvalue < 0.05:
    print("–ú–æ–¥–µ–ª–∏ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏ —Ä–∞–∑–ª–∏—á–∞—é—Ç—Å—è")
else:
    print("–ù–µ—Ç –∑–Ω–∞—á–∏–º–æ–π —Ä–∞–∑–Ω–∏—Ü—ã")

üî∑ 11. –†–∞–∑–º–µ—Ä —ç—Ñ—Ñ–µ–∫—Ç–∞

Cohen's d (–¥–ª—è t-test):

def cohens_d(x1, x2):
    """–†–∞–∑–º–µ—Ä —ç—Ñ—Ñ–µ–∫—Ç–∞ –¥–ª—è –¥–≤—É—Ö –≥—Ä—É–ø–ø"""
    mean_diff = np.mean(x1) - np.mean(x2)
    pooled_std = np.sqrt(
        (np.std(x1)**2 + np.std(x2)**2) / 2
    )
    return mean_diff / pooled_std

d = cohens_d(scores1, scores2)
print(f"Cohen's d: {d:.4f}")

# –ò–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ü–∏—è:
# |d| < 0.2: –º–∞–ª—ã–π —ç—Ñ—Ñ–µ–∫—Ç
# 0.2 < |d| < 0.8: —Å—Ä–µ–¥–Ω–∏–π
# |d| > 0.8: –±–æ–ª—å—à–æ–π

from sklearn.ensemble import RandomForestClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score

lr_scores = cross_val_score(lr, X, y, cv=10, scoring='accuracy')

# –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–æ—Ä–º–∞–ª—å–Ω–æ—Å—Ç–∏

# –í—ã–±–æ—Ä —Ç–µ—Å—Ç–∞
if p_rf > 0.05 and p_lr > 0.05:
    # –ù–æ—Ä–º–∞–ª—å–Ω—ã–µ - –∏—Å–ø–æ–ª—å–∑—É–µ–º paired t-test
    t_stat, p_val = stats.ttest_rel(rf_scores, lr_scores)
    test_name = "Paired t-test"

    # –ù–µ –Ω–æ—Ä–º–∞–ª—å–Ω—ã–µ - –∏—Å–ø–æ–ª—å–∑—É–µ–º Wilcoxon
    t_stat, p_val = wilcoxon(rf_scores, lr_scores)
    test_name = "Wilcoxon"

print(f"p-value: {p_val:.4f}")

if p_val < 0.05:
    winner = "RF" if rf_scores.mean() > lr_scores.mean() else "LR"
    print(f"{winner} significantly better")
else:
    print("No significant difference")
