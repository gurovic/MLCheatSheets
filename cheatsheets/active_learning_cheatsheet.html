def uncertainty_sampling(model, X_unlabeled, n_instances=10):
    # –ü—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ–º –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏
    proba = model.predict_proba(X_unlabeled)
    
    # –£–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å = –º–∞–∫—Å. –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å
    confidence = proba.max(axis=1)
    
    # Uncertainty = 1 - confidence
    uncertainty = 1 - confidence
    
    # –í—ã–±–∏—Ä–∞–µ–º —Ç–æ–ø-n —Å–∞–º—ã—Ö –Ω–µ—É–≤–µ—Ä–µ–Ω–Ω—ã—Ö
    query_idx = np.argsort(uncertainty)[-n_instances:]
    
    return query_idx

import numpy as np

def entropy_sampling(model, X_unlabeled, n_instances=10):
    # –ü—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ–º –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏
    proba = model.predict_proba(X_unlabeled)
    
    # –≠–Ω—Ç—Ä–æ–ø–∏—è: -Œ£ p(y) log p(y)
    entropy = -np.sum(proba * np.log(proba + 1e-10), axis=1)
    
    # –í—ã–±–∏—Ä–∞–µ–º —Ç–æ–ø-n —Å –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π —ç–Ω—Ç—Ä–æ–ø–∏–µ–π
    query_idx = np.argsort(entropy)[-n_instances:]
    
    return query_idx

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

üî∑ 7. Margin Sampling

def margin_sampling(model, X_unlabeled, n_instances=10):
    # –ü—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ–º –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏
    proba = model.predict_proba(X_unlabeled)
    
    # –°–æ—Ä—Ç–∏—Ä—É–µ–º –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏
    proba_sorted = np.sort(proba, axis=1)
    
    # Margin = —Ä–∞–∑–Ω–∏—Ü–∞ —Ç–æ–ø-2 –∫–ª–∞—Å—Å–æ–≤
    margin = proba_sorted[:, -1] - proba_sorted[:, -2]
    
    # –í—ã–±–∏—Ä–∞–µ–º —Ç–æ–ø-n —Å –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–º margin
    query_idx = np.argsort(margin)[:n_instances]
    
    return query_idx

    RandomForestClassifier(),
    GradientBoostingClassifier(),
    LogisticRegression()
]

# –í—ã–±–æ—Ä –ø–æ —Ä–∞–∑–Ω–æ–≥–ª–∞—Å–∏—è–º
def query_by_committee(committee, X_unlabeled, n_instances=10):
    # –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –≤—Å–µ—Ö –º–æ–¥–µ–ª–µ–π
    predictions = np.array([
        model.predict(X_unlabeled) for model in committee
    ])
    
    # –†–∞–∑–Ω–æ–≥–ª–∞—Å–∏—è = —Ä–∞–∑–Ω–æ–æ–±—Ä–∞–∑–∏–µ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π
    # Vote entropy
    vote_entropy = []
    for i in range(len(X_unlabeled)):
        votes = predictions[:, i]
        unique, counts = np.unique(votes, return_counts=True)
        proba = counts / len(committee)
        entropy = -np.sum(proba * np.log(proba + 1e-10))
        vote_entropy.append(entropy)
    
    vote_entropy = np.array(vote_entropy)
    
    # –í—ã–±–∏—Ä–∞–µ–º —Ç–æ–ø-n —Å –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–º–∏ —Ä–∞–∑–Ω–æ–≥–ª–∞—Å–∏—è–º–∏
    query_idx = np.argsort(vote_entropy)[-n_instances:]
    return query_idx

from sklearn.ensemble import RandomForestClassifier
import numpy as np

# –ù–∞—á–∞–ª—å–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
labeled_idx = np.random.choice(len(X_train), 100, replace=False)
X_labeled = X_train[labeled_idx]
y_labeled = y_train[labeled_idx]

unlabeled_idx = np.setdiff1d(range(len(X_train)), labeled_idx)
X_unlabeled = X_train[unlabeled_idx]

# –ú–æ–¥–µ–ª—å
model = RandomForestClassifier()
model.fit(X_labeled, y_labeled)

# Active learning —Ü–∏–∫–ª
for iteration in range(10):
    # Query strategy (uncertainty sampling)
    proba = model.predict_proba(X_unlabeled)
    uncertainty = 1 - proba.max(axis=1)
    query_idx = np.argsort(uncertainty)[-20:]  # —Ç–æ–ø-20
    
    # –°–∏–º—É–ª—è—Ü–∏—è —Ä–∞–∑–º–µ—Ç–∫–∏ (–≤ —Ä–µ–∞–ª—å–Ω–æ—Å—Ç–∏ - —á–µ–ª–æ–≤–µ–∫)
