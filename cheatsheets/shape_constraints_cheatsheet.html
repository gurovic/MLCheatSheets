import torch.nn as nn

class MonotonicNN(nn.Module):
    def __init__(self, input_dim):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, 32)
        self.fc2 = nn.Linear(32, 1)
    
    def forward(self, x):
        # –û–±–µ—Å–ø–µ—á–∏–≤–∞–µ–º –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ –≤–µ—Å–∞

        return self.fc2(h)

import numpy as np

def check_monotonicity(model, X, feature_idx, 
                       direction='increasing'):
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç –º–æ–Ω–æ—Ç–æ–Ω–Ω–æ—Å—Ç—å –ø–æ –ø—Ä–∏–∑–Ω–∞–∫—É"""
    violations = 0
    
    for i in range(len(X)):
        x = X[i].copy()
        original_value = x[feature_idx]
        original_pred = model.predict([x])[0]
        
        # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º –ø—Ä–∏–∑–Ω–∞–∫
        x[feature_idx] = original_value * 1.1
        new_pred = model.predict([x])[0]
        
        if direction == 'increasing':
            if new_pred < original_pred:
                violations += 1
        else:  # decreasing
            if new_pred > original_pred:
                violations += 1
    
    return violations / len(X)

üî∑ 10. –ú–æ–Ω–æ—Ç–æ–Ω–Ω–æ—Å—Ç—å –≤ –ª–∏–Ω–µ–π–Ω—ã—Ö –º–æ–¥–µ–ª—è—Ö

from sklearn.linear_model import Ridge
from scipy.optimize import minimize

def monotonic_linear_regression(X, y, 
                                 monotone_features):
    """Linear regression —Å —Ñ–æ—Ä–º—ã–º–∏ constraints"""
    def objective(w):
        pred = X @ w
        return np.mean((pred - y)**2)
    
    # Constraints: –≤–µ—Å–∞ —Ñ–æ—Ä–º—ã—Ö –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ >= 0
    constraints = [
        {'type': 'ineq', 'fun': lambda w: w[i]}
        for i in monotone_features
    ]
    
    result = minimize(
        objective, 
        x0=np.zeros(X.shape[1]),
        constraints=constraints
    )
    return result.x
