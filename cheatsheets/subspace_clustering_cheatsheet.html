def clique(X, xi, tau):
    """
    X: –¥–∞–Ω–Ω—ã–µ

    for dim in range(d):
        _, edges = np.histogram(X[:, dim], bins=xi)
        bins.append(edges)
    
    # –ù–∞–π—Ç–∏ –ø–ª–æ—Ç–Ω—ã–µ –µ–¥–∏–Ω–∏—Ü—ã –≤ 1D

        counts = np.histogram(X[:, dim], bins=bins[dim])[0]
        dense = counts >= tau
        dense_units[dim] = np.where(dense)[0]
    
    # –û–±—ä–µ–¥–∏–Ω—è—Ç—å –≤ k-–º–µ—Ä–Ω—ã–µ –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞
    # (—É–ø—Ä–æ—â—ë–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è)
    return dense_units

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

# –ü—Ä–æ—Å—Ç–∞—è –≤–µ—Ä—Å–∏—è subspace clustering
from sklearn.cluster import KMeans
from sklearn.feature_selection import mutual_info_classif

def simple_subspace_clustering(X, k, l):
    """
    k: —á–∏—Å–ª–æ –∫–ª–∞—Å—Ç–µ—Ä–æ–≤
    l: —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤
    """
    n, d = X.shape
    
    # 1. –ù–∞—á–∞–ª—å–Ω–∞—è –∫–ª–∞—Å—Ç–µ—Ä–∏–∑–∞—Ü–∏—è
    kmeans = KMeans(n_clusters=k, random_state=42)
    labels = kmeans.fit_predict(X)
    
    # 2. –î–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Ç–µ—Ä–∞ –Ω–∞–π—Ç–∏ l –ª—É—á—à–∏—Ö –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
    subspaces = {}
    for cluster_id in range(k):
        mask = labels == cluster_id
        X_cluster = X[mask]
        
        # –í—ã—á–∏—Å–ª–∏—Ç—å –≤–∞–∂–Ω–æ—Å—Ç—å –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
        # (–Ω–∞–ø—Ä–∏–º–µ—Ä, variance –∏–ª–∏ mutual information)
        importance = X_cluster.var(axis=0)
        top_dims = np.argsort(importance)[-l:]
        subspaces[cluster_id] = top_dims
    
    # 3. –ü–µ—Ä–µ–Ω–∞–∑–Ω–∞—á–∏—Ç—å —Ç–æ—á–∫–∏ —Å —É—á—ë—Ç–æ–º –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤
    new_labels = reassign_subspace(X, kmeans.cluster_centers_, 
                                    subspaces)
    
    return new_labels, subspaces

# –ö–∞—á–µ—Å—Ç–≤–æ subspace clustering
def subspace_quality(X, labels, subspaces):
    """–û—Ü–µ–Ω–∫–∞ –∫–∞—á–µ—Å—Ç–≤–∞"""
    scores = []
    for cluster_id in np.unique(labels):
        mask = labels == cluster_id
        X_cluster = X[mask]
        dims = subspaces[cluster_id]
        
        # –ü—Ä–æ–µ–∫—Ü–∏—è –Ω–∞ –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ
        X_proj = X_cluster[:, dims]
        
        # Compactness –≤ –ø–æ–¥–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ
        center = X_proj.mean(axis=0)
        dist = np.linalg.norm(X_proj - center, axis=1)
        scores.append(dist.mean())
    
    return np.mean(scores)

score = subspace_quality(X, labels, subspaces)
print(f"Subspace quality: {score:.3f}")
