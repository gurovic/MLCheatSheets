from lime import lime_image
from PIL import Image

def predict_fn(images):
    model.eval()
    batch = torch.stack([
        preprocess(Image.fromarray(img.astype('uint8')))
        for img in images
    ])
    
    with torch.no_grad():
        outputs = model(batch.to(device))
        probs = torch.nn.functional.softmax(
            outputs, dim=1
        )
    return probs.cpu().numpy()

    np.array(image),
    predict_fn,
    top_labels=5,

    num_samples=100
