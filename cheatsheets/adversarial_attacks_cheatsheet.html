üî∑ 3. FGSM (Fast Gradient Sign Method)

–°–∞–º–∞—è –ø—Ä–æ—Å—Ç–∞—è –∏ –±—ã—Å—Ç—Ä–∞—è –∞—Ç–∞–∫–∞. –ò—Å–ø–æ–ª—å–∑—É–µ—Ç –∑–Ω–∞–∫ –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞ —Ñ—É–Ω–∫—Ü–∏–∏ –ø–æ—Ç–µ—Ä—å.

import torch
import torch.nn.functional as F

def fgsm_attack(image, epsilon, data_grad):
    # –ó–Ω–∞–∫ –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞
    sign_data_grad = data_grad.sign()
    # –°–æ–∑–¥–∞–Ω–∏–µ –≤–æ–∑–º—É—â—ë–Ω–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    perturbed_image = image + epsilon * sign_data_grad
    # –û–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–µ –¥–∏–∞–ø–∞–∑–æ–Ω–∞ [0, 1]
    perturbed_image = torch.clamp(perturbed_image, 0, 1)
    return perturbed_image

üî∑ 4. PGD (Projected Gradient Descent)

–ò—Ç–µ—Ä–∞—Ç–∏–≤–Ω–∞—è –≤–µ—Ä—Å–∏—è FGSM. –°–∏–ª—å–Ω–µ–µ –∏ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–µ–µ.

def pgd_attack(model, images, labels, epsilon=0.3, alpha=0.01, num_iter=40):
    # –ö–æ–ø–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    perturbed_images = images.clone().detach()
    perturbed_images.requires_grad = True
    
    for i in range(num_iter):
        outputs = model(perturbed_images)
        loss = F.cross_entropy(outputs, labels)
        
        model.zero_grad()
        loss.backward()
        
        # –®–∞–≥ –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞
        data_grad = perturbed_images.grad.data
        perturbed_images = perturbed_images.detach() + alpha * data_grad.sign()
        
        # –ü—Ä–æ–µ—Ü–∏—Ä–æ–≤–∞–Ω–∏–µ –≤ epsilon-—à–∞—Ä
        eta = torch.clamp(perturbed_images - images, -epsilon, epsilon)
        perturbed_images = torch.clamp(images + eta, 0, 1).detach()
        perturbed_images.requires_grad = True
    
    return perturbed_images

# PGD ‚Äî –æ–¥–Ω–∞ –∏–∑ —Å–∞–º—ã—Ö —Å–∏–ª—å–Ω—ã—Ö white-box –∞—Ç–∞–∫

üî∑ 5. C&W Attack (Carlini & Wagner)

–û–ø—Ç–∏–º–∏–∑–∞—Ü–∏–æ–Ω–Ω–∞—è –∞—Ç–∞–∫–∞, –Ω–∞—Ö–æ–¥–∏—Ç –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –≤–æ–∑–º—É—â–µ–Ω–∏–µ.

# –£–ø—Ä–æ—â—ë–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è C&W L2
def cw_l2_attack(model, image, target, c=1.0, kappa=0, max_iter=1000, lr=0.01):
    # –ü–µ—Ä–µ–º–µ–Ω–Ω—ã–µ
    w = torch.zeros_like(image, requires_grad=True)
    optimizer = torch.optim.Adam([w], lr=lr)
    
    for step in range(max_iter):
        # –°–æ–∑–¥–∞–Ω–∏–µ –≤–æ–∑–º—É—â—ë–Ω–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        perturbed = 0.5 * (torch.tanh(w) + 1)
        
        # –í—ã—Ö–æ–¥ –º–æ–¥–µ–ª–∏
        output = model(perturbed)
        
        # Loss C&W
        real = output[:, target]
        other = torch.max((1 - F.one_hot(target, output.shape[1])) * output, dim=1)[0]
        f_loss = torch.clamp(real - other + kappa, min=0)
        
        # L2 –Ω–æ—Ä–º–∞ –≤–æ–∑–º—É—â–µ–Ω–∏—è
        l2_loss = torch.norm(perturbed - image, p=2)
        
        # –û–±—â–∞—è —Ñ—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å
        loss = l2_loss + c * f_loss
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    
    return 0.5 * (torch.tanh(w) + 1)

# C&W –Ω–∞—Ö–æ–¥–∏—Ç –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –≤–æ–∑–º—É—â–µ–Ω–∏–µ, —Å–ª–æ–∂–Ω–µ–µ –¥–µ—Ç–µ–∫—Ç–∏—Ä–æ–≤–∞—Ç—å

üî∑ 6. DeepFool

–ù–∞—Ö–æ–¥–∏—Ç –∫—Ä–∞—Ç—á–∞–π—à–∏–π –ø—É—Ç—å –∫ –≥—Ä–∞–Ω–∏—Ü–µ —Ä–µ—à–µ–Ω–∏—è.

def deepfool(model, image, num_classes=10, max_iter=50, overshoot=0.02):
    f_image = model(image).detach()
    I = f_image.argmax().item()
    
    input_shape = image.shape
    pert_image = image.clone()
    
    w = torch.zeros(input_shape)
    r_tot = torch.zeros(input_shape)
    
    for iter in range(max_iter):
        pert_image.requires_grad = True
        fs = model(pert_image)
        
        # –ì—Ä–∞–¥–∏–µ–Ω—Ç—ã
        grads = torch.zeros((num_classes,) + input_shape[1:])
        for k in range(num_classes):
            model.zero_grad()
            fs[0, k].backward(retain_graph=True)
            grads[k] = pert_image.grad.data.clone()
        
        # –ù–∞—Ö–æ–¥–∏–º –±–ª–∏–∂–∞–π—à—É—é –≥—Ä–∞–Ω–∏—Ü—É
        pert = float('inf')
        for k in range(num_classes):
            if k == I:
                continue
            
            w_k = grads[k] - grads[I]
            f_k = fs[0, k] - fs[0, I]
            
            pert_k = abs(f_k) / torch.norm(w_k.flatten())
            
            if pert_k < pert:
                pert = pert_k
                w = w_k
        
        # –î–æ–±–∞–≤–ª—è–µ–º –≤–æ–∑–º—É—â–µ–Ω–∏–µ
        r_i = (pert + 1e-4) * w / torch.norm(w)
        r_tot += r_i
        
        pert_image = image + (1 + overshoot) * r_tot
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–∏–µ –∫–ª–∞—Å—Å–∞
        pred = model(pert_image).argmax().item()
        if pred != I:
            break
    
    return r_tot, pert_image, pred

def adversarial_training(model, train_loader, epochs=10, epsilon=0.1):
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    
    for epoch in range(epochs):
        for images, labels in train_loader:
            # –û–±—ã—á–Ω–æ–µ –æ–±—É—á–µ–Ω–∏–µ
            outputs = model(images)
            loss_clean = F.cross_entropy(outputs, labels)
            
            # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è adversarial –ø—Ä–∏–º–µ—Ä–æ–≤
            images.requires_grad = True
            outputs = model(images)
            loss = F.cross_entropy(outputs, labels)
            model.zero_grad()
            loss.backward()
            
            # FGSM –∞—Ç–∞–∫–∞
            perturbed = fgsm_attack(images, epsilon, images.grad.data)
            
            # –û–±—É—á–µ–Ω–∏–µ –Ω–∞ adversarial –ø—Ä–∏–º–µ—Ä–∞—Ö
            outputs_adv = model(perturbed)
            loss_adv = F.cross_entropy(outputs_adv, labels)
            
            # –ö–æ–º–±–∏–Ω–∏—Ä–æ–≤–∞–Ω–Ω–∞—è loss
            total_loss = 0.5 * loss_clean + 0.5 * loss_adv
            
            optimizer.zero_grad()
            total_loss.backward()
            optimizer.step()
    
    return model

def detect_by_reconstruction(autoencoder, image, threshold=0.1):

    error = F.mse_loss(image, reconstructed).item()

def detect_by_stats(image, clean_stats):

# Randomized Smoothing (—É–ø—Ä–æ—â—ë–Ω–Ω–æ)
def randomized_smoothing_predict(model, image, num_samples=100, sigma=0.25):
    predictions = []
    
    for _ in range(num_samples):
        # –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —à—É–º–∞
        noisy = image + torch.randn_like(image) * sigma
        output = model(noisy)
        predictions.append(output.argmax().item())
    
    # –ú–∞–∂–æ—Ä–∏—Ç–∞—Ä–Ω–æ–µ –≥–æ–ª–æ—Å–æ–≤–∞–Ω–∏–µ
    from collections import Counter
    most_common = Counter(predictions).most_common(1)[0]
    return most_common[0], most_common[1] / num_samples

def evaluate_robustness(model, test_loader, attack_fn, epsilon=0.1):
    correct_clean = 0
    correct_adv = 0
    total = 0
    
    for images, labels in test_loader:
        # –ß–∏—Å—Ç—ã–µ –¥–∞–Ω–Ω—ã–µ
        outputs_clean = model(images)
        correct_clean += (outputs_clean.argmax(1) == labels).sum().item()
        
        # Adversarial –¥–∞–Ω–Ω—ã–µ
        images_adv = attack_fn(model, images, labels, epsilon)
        outputs_adv = model(images_adv)
        correct_adv += (outputs_adv.argmax(1) == labels).sum().item()
        
        total += labels.size(0)
    
    clean_acc = 100 * correct_clean / total
    robust_acc = 100 * correct_adv / total
    
    print(f"Clean Accuracy: {clean_acc:.2f}%")
    print(f"Robust Accuracy: {robust_acc:.2f}%")
    print(f"Attack Success Rate: {100 - robust_acc:.2f}%")
