ARIMA(p, d, q):
p = –ø–æ—Ä—è–¥–æ–∫ –∞–≤—Ç–æ—Ä–µ–≥—Ä–µ—Å—Å–∏–∏ (AR)
d = –ø–æ—Ä—è–¥–æ–∫ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏—Ä–æ–≤–∞–Ω–∏—è (I)

from statsmodels.tsa.stattools import adfuller, kpss
import pandas as pd

# Augmented Dickey-Fuller —Ç–µ—Å—Ç
def check_stationarity(series):
    # ADF —Ç–µ—Å—Ç (H0: –Ω–µ—Å—Ç–∞—Ü–∏–æ–Ω–∞—Ä–µ–Ω)
    adf_result = adfuller(series)
    print(f'ADF Statistic: {adf_result[0]:.4f}')
    print(f'p-value: {adf_result[1]:.4f}')
    
    # KPSS —Ç–µ—Å—Ç (H0: —Å—Ç–∞—Ü–∏–æ–Ω–∞—Ä–µ–Ω)
    kpss_result = kpss(series)
    print(f'KPSS Statistic: {kpss_result[0]:.4f}')
    print(f'p-value: {kpss_result[1]:.4f}')
    
    if adf_result[1] < 0.05:
        print("‚úÖ –†—è–¥ —Å—Ç–∞—Ü–∏–æ–Ω–∞—Ä–µ–Ω (ADF)")
    else:
        print("‚ùå –†—è–¥ –Ω–µ—Å—Ç–∞—Ü–∏–æ–Ω–∞—Ä–µ–Ω (ADF)")

check_stationarity(df['value'])
