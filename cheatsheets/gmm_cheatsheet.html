import matplotlib.pyplot as plt
from matplotlib.patches import Ellipse

def plot_gmm(gmm, X, labels):
    plt.figure(figsize=(10, 6))
    
    # –¢–æ—á–∫–∏
    plt.scatter(X[:, 0], X[:, 1], c=labels, 
               cmap='viridis', alpha=0.6)
    
    # –≠–ª–ª–∏–ø—Å—ã –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Ç–µ—Ä–∞
    for i in range(gmm.n_components):
        mean = gmm.means_[i]
        covar = gmm.covariances_[i]
        
        # –°–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –∏ –≤–µ–∫—Ç–æ—Ä—ã
        eigenvalues, eigenvectors = np.linalg.eigh(covar)
        angle = np.degrees(np.arctan2(eigenvectors[1, 0], 
                                      eigenvectors[0, 0]))
        width, height = 2 * np.sqrt(eigenvalues)
        
        # –†–∏—Å—É–µ–º —ç–ª–ª–∏–ø—Å (2œÉ)
        ellipse = Ellipse(mean, width, height, 
                         angle=angle, alpha=0.3)
        plt.gca().add_patch(ellipse)
        
        # –¶–µ–Ω—Ç—Ä
        plt.plot(mean[0], mean[1], 'kx', markersize=10)
    
    plt.title('GMM –∫–ª–∞—Å—Ç–µ—Ä–∏–∑–∞—Ü–∏—è —Å —ç–ª–ª–∏–ø—Å–∞–º–∏')
    plt.xlabel('–ü—Ä–∏–∑–Ω–∞–∫ 1')
    plt.ylabel('–ü—Ä–∏–∑–Ω–∞–∫ 2')
    plt.show()

plot_gmm(gmm, X_scaled, labels)
