–ü—Ä–æ–±–ª–µ–º–∞: –ù–µ–ª—å–∑—è —Ö—Ä–∞–Ω–∏—Ç—å –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–µ –∞–∫—Ç–∏–≤–∞—Ü–∏–∏ (–∏—Ö –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ –º–Ω–æ–≥–æ!)

–†–µ—à–µ–Ω–∏–µ: Adjoint sensitivity method

–í–º–µ—Å—Ç–æ backprop —á–µ—Ä–µ–∑ ODE solver:

1. Forward pass: —Ä–µ—à–∏—Ç—å ODE
   h(T) = ODESolve(h(0), f, [0,T])

2. Backward pass: —Ä–µ—à–∏—Ç—å augmented ODE
   [dL/dh(0), dL/dŒ∏] = ODESolve([dL/dh(T), 0], g, [T,0])

–≥–¥–µ g - adjoint dynamics

–ü—Ä–µ–∏–º—É—â–µ—Å—Ç–≤–æ: O(1) –ø–∞–º—è—Ç—å!

import torch
import torch.nn as nn
from torchdiffeq import odeint

class ODEFunc(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(dim, 64),
            nn.Tanh(),
            nn.Linear(64, dim)
        )
    
    def forward(self, t, h):
        return self.net(h)

class NeuralODE(nn.Module):
    def __init__(self, func):
        super().__init__()
        self.func = func
    
    def forward(self, x):
        # –ò–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞—Ç—å –æ—Ç t=0 –¥–æ t=1
        t = torch.tensor([0., 1.])
        h = odeint(self.func, x, t, method='dopri5')
        return h[1]  # –í–µ—Ä–Ω—É—Ç—å h(1)

–ü—Ä–µ–∏–º—É—â–µ—Å—Ç–≤–æ: –ù–µ –Ω—É–∂–Ω–∞ –æ–±—Ä–∞—Ç–∏–º–æ—Å—Ç—å –∫–∞–∂–¥–æ–≥–æ —Å–ª–æ—è!

# –ü—Ä–∏–º–µ—Ä
class CNF(nn.Module):
    def __init__(self, func, T=1.0):
        super().__init__()
        self.func = func
        self.T = T
    
    def forward(self, x):
        # Augment with log determinant
        h_aug = torch.cat([x, torch.zeros(x.shape[0], 1)], 1)
        
        # Integrate
        t = torch.tensor([0., self.T])
        h_T = odeint(self.func, h_aug, t)
        
        z, log_det = h_T[1][:, :-1], h_T[1][:, -1]
        return z, log_det

class TimeDependentODEFunc(nn.Module):
    def __init__(self, dim):
        super().__init__()
        # –ü–∞—Ä–∞–º–µ—Ç—Ä–∏–∑–∞—Ü–∏—è –∑–∞–≤–∏—Å–∏—Ç –æ—Ç t

            nn.Linear(dim + 1, 64),  # +1 –¥–ª—è –≤—Ä–µ–º–µ–Ω–∏
            nn.Tanh(),
            nn.Linear(64, dim)
        )
    
    def forward(self, t, h):
        # Concatenate time
        t_vec = torch.ones(h.shape[0], 1) * t
        h_t = torch.cat([h, t_vec], dim=1)
        return self.net(h_t)
