üî∑ 1. –°—É—Ç—å –º–µ—Ç–æ–¥–∞

Bagging (Bootstrap AGGregatING) ‚Äî –º–µ—Ç–æ–¥ –∞–Ω—Å–∞–º–±–ª–∏—Ä–æ–≤–∞–Ω–∏—è, –∫–æ—Ç–æ—Ä—ã–π —Å–æ–∑–¥–∞—ë—Ç –Ω–µ—Å–∫–æ–ª—å–∫–æ –≤–µ—Ä—Å–∏–π –æ–±—É—á–∞—é—â–µ–π –≤—ã–±–æ—Ä–∫–∏ —Å –ø–æ–º–æ—â—å—é bootstrap-—Å—ç–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏—è.

  • –ò–¥–µ—è: –æ–±—É—á–∏—Ç—å –º–Ω–æ–≥–æ –º–æ–¥–µ–ª–µ–π –Ω–∞ —Ä–∞–∑–Ω—ã—Ö –ø–æ–¥–≤—ã–±–æ—Ä–∫–∞—Ö
  • Bootstrap: —Å–ª—É—á–∞–π–Ω–∞—è –≤—ã–±–æ—Ä–∫–∞ —Å –≤–æ–∑–≤—Ä–∞—â–µ–Ω–∏–µ–º
  • –ê–≥—Ä–µ–≥–∞—Ü–∏—è: —É—Å—Ä–µ–¥–Ω–µ–Ω–∏–µ (—Ä–µ–≥—Ä–µ—Å—Å–∏—è) –∏–ª–∏ –≥–æ–ª–æ—Å–æ–≤–∞–Ω–∏–µ (–∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è)
  • –¶–µ–ª—å: —Å–Ω–∏–∑–∏—Ç—å –¥–∏—Å–ø–µ—Ä—Å–∏—é –∏ –ø–µ—Ä–µ–æ–±—É—á–µ–Ω–∏–µ
  • –≠—Ñ—Ñ–µ–∫—Ç: –±–æ–ª–µ–µ —Å—Ç–∞–±–∏–ª—å–Ω—ã–µ –∏ —Ç–æ—á–Ω—ã–µ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è
–ë—ç–≥–≥–∏–Ω–≥ –æ—Å–æ–±–µ–Ω–Ω–æ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–µ–Ω –¥–ª—è –Ω–µ—Å—Ç–∞–±–∏–ª—å–Ω—ã—Ö –º–æ–¥–µ–ª–µ–π (–¥–µ—Ä–µ–≤—å—è —Ä–µ—à–µ–Ω–∏–π), –≥–¥–µ –Ω–µ–±–æ–ª—å—à–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö —Å–∏–ª—å–Ω–æ –º–µ–Ω—è–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç.

  2. –û–±—É—á–∏—Ç—å –±–∞–∑–æ–≤—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º –Ω–∞ –∫–∞–∂–¥–æ–π –≤—ã–±–æ—Ä–∫–µ
  3. –î–ª—è —Ä–µ–≥—Ä–µ—Å—Å–∏–∏: —É—Å—Ä–µ–¥–Ω–∏—Ç—å –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –≤—Å–µ—Ö –º–æ–¥–µ–ª–µ–π
  4. –î–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏: –≤—ã–±—Ä–∞—Ç—å –∫–ª–∞—Å—Å –ø–æ –º–∞–∂–æ—Ä–∏—Ç–∞—Ä–Ω–æ–º—É –≥–æ–ª–æ—Å–æ–≤–∞–Ω–∏—é

    X, y, test_size=0.2, random_state=42

    estimator=DecisionTreeClassifier(),
    n_estimators=100,           # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –º–æ–¥–µ–ª–µ–π
    max_samples=1.0,            # –†–∞–∑–º–µ—Ä –∫–∞–∂–¥–æ–π –≤—ã–±–æ—Ä–∫–∏
    max_features=1.0,           # –î–æ–ª—è –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
    bootstrap=True,             # –ò—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å bootstrap
    bootstrap_features=False,   # –ù–µ —Å—ç–º–ø–ª–∏—Ä–æ–≤–∞—Ç—å –ø—Ä–∏–∑–Ω–∞–∫–∏
    oob_score=True,            # Out-of-bag –æ—Ü–µ–Ω–∫–∞
    random_state=42,
    n_jobs=-1                  # –ü–∞—Ä–∞–ª–ª–µ–ª—å–Ω–æ—Å—Ç—å
)

# –û–±—É—á–µ–Ω–∏–µ
bagging.fit(X_train, y_train)

# –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ

print(f"Accuracy: {accuracy_score(y_test, y_pred):.4f}")

# Out-of-bag –æ—Ü–µ–Ω–∫–∞
print(f"OOB Score: {bagging.oob_score_:.4f}")

# –û–¥–Ω–æ –¥–µ—Ä–µ–≤–æ

    single_tree, X, y, cv=5, scoring='accuracy'

    DecisionTreeClassifier(),
    n_estimators=100,
    random_state=42

)
