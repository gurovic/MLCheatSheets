# –§–æ—Ä–º—É–ª–∞ –≤—Ä—É—á–Ω—É—é
def ses_manual(y, alpha=0.2):
    result = [y[0]]  # –ù–∞—á–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    for i in range(1, len(y)):
        result.append(alpha * y[i] + (1 - alpha) * result[i-1])
    return np.array(result)

smoothed = ses_manual(y, alpha=0.2)

# Cross-validation –¥–ª—è –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ä—è–¥–æ–≤

for train_idx, test_idx in tscv.split(y):
    train, test = y[train_idx], y[test_idx]
    
    model = ExponentialSmoothing(train, trend='add', seasonal='add', 
                                  seasonal_periods=12)
    fitted = model.fit()
    forecast = fitted.forecast(len(test))
    
    mse = mean_squared_error(test, forecast)
    scores.append(mse)

print(f"Average CV MSE: {np.mean(scores):.2f}")
